// ==UserScript==
// @name         YouTube Volume Slider 2.0
// @namespace    http://tampermonkey.net/
// @version      2.0
// @description  Compact in-bar volume indicator that expands into a wide YouTube volume slider.
// @author       Gimbuhh (Made using AI)
// @icon         https://www.youtube.com/favicon.ico
// @match        https://www.youtube.com/*
// @run-at       document-idle
// @grant        none
// ==/UserScript==

(function () {
    'use strict';

    const OVERLAY_ID = 'tm-volume-slider-overlay';
    const SLIDER_ID = 'tm-volume-slider-range';
    const VALUE_LABEL_ID = 'tm-volume-slider-value';
    const SETTINGS_STYLE_ID = 'tm-volume-settings-style';
    const STORAGE_KEY = 'tm-yt-volume';
    const VOLUME_MODE_KEY = 'tm-yt-volume-slider-mode';
    const SLIDER_LOCATION_KEY = 'tm-yt-volume-slider-location';
    const REPLACE_NATIVE_PLACEMENT_KEY = 'tm-yt-volume-slider-replace-placement';
    const SNAP_TO_5_KEY = 'tm-yt-volume-slider-snap-to-5';
    const ALWAYS_EXPANDED_KEY = 'tm-yt-volume-slider-always-expanded';
    const OVERLAY_ENABLED_KEY = 'tm-yt-overlay-enabled';
    const REPLACE_NATIVE_KEY = 'tm-yt-replace-native-volume';
    const STORAGE_WRITE_DEBOUNCE_MS = 150;
    const NAV_REATTACH_DELAY_MS = 700;
    const NAV_DEBOUNCE_MS = 180;
    const USER_SETTINGS = {
        // Volume slider mode: 'saved', 'off', 'on', or 'replace-native'. Default: 'saved'
        volumeSliderMode: 'saved',
        // Slider location: 'saved', 'controls', or 'video'. Default: 'saved'
        sliderLocation: 'saved',
        // Replace-native placement: 'saved', 'native', or 'custom'. Default: 'saved'
        replaceNativePlacement: 'saved',
        // Snap slider movement to 5% steps: 'saved', true, or false. Default: 'saved'
        snapToFive: 'saved',
        // Keep the volume pill expanded: 'saved', true, or false. Default: 'saved'
        alwaysExpanded: 'saved',
        // On-video slider opacity when unfocused (0-100 as a percentage). Default: 45
        overlayOpacityUnfocused: 45,
        // On-video slider opacity when focused/hovered (0-100 as a percentage). Default: 95
        overlayOpacityFocused: 95
    };

    let cachedVideo = null;
    let cachedYtPlayer = null;
    let saveVolumeTimer = 0;
    let navReattachTimer = 0;
    let navDebounceTimer = 0;

    // -------------------------------
    // Overlay + YouTube Settings State
    // -------------------------------

    function normalizeVolumeSliderMode(mode) {
        return ['off', 'on', 'replace-native'].includes(mode) ? mode : null;
    }

    function normalizeReplaceNativePlacement(placement) {
        return ['native', 'custom'].includes(placement) ? placement : null;
    }

    function normalizeSliderLocation(location) {
        return ['controls', 'video'].includes(location) ? location : null;
    }

    function normalizeBooleanSetting(value) {
        if (value === true || value === 'true') return true;
        if (value === false || value === 'false') return false;
        return null;
    }

    function getSavedVolumeSliderMode() {
        try {
            const savedMode = normalizeVolumeSliderMode(localStorage.getItem(VOLUME_MODE_KEY));
            if (savedMode) return savedMode;

            const savedEnabled = localStorage.getItem(OVERLAY_ENABLED_KEY);
            const savedReplaceNative = localStorage.getItem(REPLACE_NATIVE_KEY) === 'true';
            if (savedEnabled === 'false') return 'off';
            if (savedReplaceNative) return 'replace-native';
        } catch (e) { /* ignore storage errors */ }
        return 'on';
    }

    function getVolumeSliderMode() {
        const overrideMode = normalizeVolumeSliderMode(USER_SETTINGS.volumeSliderMode);
        return overrideMode || getSavedVolumeSliderMode();
    }

    function getReplaceNativePlacement() {
        const overridePlacement = normalizeReplaceNativePlacement(USER_SETTINGS.replaceNativePlacement);
        if (overridePlacement) return overridePlacement;
        try {
            return normalizeReplaceNativePlacement(localStorage.getItem(REPLACE_NATIVE_PLACEMENT_KEY)) || 'native';
        } catch (e) {
            return 'native';
        }
    }

    function getSliderLocation() {
        const overrideLocation = normalizeSliderLocation(USER_SETTINGS.sliderLocation);
        if (overrideLocation) return overrideLocation;
        try {
            return normalizeSliderLocation(localStorage.getItem(SLIDER_LOCATION_KEY)) || 'controls';
        } catch (e) {
            return 'controls';
        }
    }

    function isSliderOnVideo() {
        return getSliderLocation() === 'video';
    }

    function setSliderLocation(location) {
        const normalizedLocation = normalizeSliderLocation(location) || 'controls';
        try {
            localStorage.setItem(SLIDER_LOCATION_KEY, normalizedLocation);
        } catch (e) { /* ignore storage errors */ }
        attachSliderIfPossible();
        applyNativeVolumeVisibility();
    }

    function setReplaceNativePlacement(placement) {
        const normalizedPlacement = normalizeReplaceNativePlacement(placement) || 'native';
        try {
            localStorage.setItem(REPLACE_NATIVE_PLACEMENT_KEY, normalizedPlacement);
        } catch (e) { /* ignore storage errors */ }
        attachSliderIfPossible();
        applyNativeVolumeVisibility();
    }

    function isSnapTo5Enabled() {
        const override = normalizeBooleanSetting(USER_SETTINGS.snapToFive);
        if (override !== null) return override;
        try {
            return localStorage.getItem(SNAP_TO_5_KEY) === 'true';
        } catch (e) {
            return false;
        }
    }

    function setSnapTo5Enabled(enabled) {
        try {
            localStorage.setItem(SNAP_TO_5_KEY, enabled ? 'true' : 'false');
        } catch (e) { /* ignore storage errors */ }
    }

    function isAlwaysExpandedEnabled() {
        const override = normalizeBooleanSetting(USER_SETTINGS.alwaysExpanded);
        if (override !== null) return override;
        try {
            return localStorage.getItem(ALWAYS_EXPANDED_KEY) === 'true';
        } catch (e) {
            return false;
        }
    }

    function setAlwaysExpandedEnabled(enabled) {
        try {
            localStorage.setItem(ALWAYS_EXPANDED_KEY, enabled ? 'true' : 'false');
        } catch (e) { /* ignore storage errors */ }

        const overlay = document.getElementById(OVERLAY_ID);
        if (!overlay) return;
        if (enabled) {
            clearExpandedHold(overlay);
            setOverlayExpanded(overlay, true);
            return;
        }
        clearExpandedHold(overlay);
        collapseOverlayIfIdle(overlay, true);
    }

    function normalizeOpacityPercent(value, fallback) {
        const num = Number(value);
        if (!Number.isFinite(num)) return fallback;
        return Math.min(100, Math.max(0, num));
    }

    function getOverlayOpacityValue(focused) {
        if (!isSliderOnVideo()) return '1';
        const fallback = focused ? 95 : 45;
        const setting = focused ? USER_SETTINGS.overlayOpacityFocused : USER_SETTINGS.overlayOpacityUnfocused;
        return String(normalizeOpacityPercent(setting, fallback) / 100);
    }

    function isOverlayInteractionFocused(overlay) {
        return overlay?.dataset.tmDragging === 'true' ||
            overlay?.matches?.(':hover') ||
            overlay?.contains?.(document.activeElement);
    }

    function updateOverlayOpacity(overlay) {
        if (!overlay) return;
        overlay.style.opacity = getOverlayOpacityValue(isOverlayInteractionFocused(overlay));
    }

    function setVolumeSliderMode(mode) {
        const normalizedMode = normalizeVolumeSliderMode(mode) || 'on';
        try {
            localStorage.setItem(VOLUME_MODE_KEY, normalizedMode);
            localStorage.setItem(OVERLAY_ENABLED_KEY, normalizedMode === 'off' ? 'false' : 'true');
            localStorage.setItem(REPLACE_NATIVE_KEY, normalizedMode === 'replace-native' ? 'true' : 'false');
        } catch (e) { /* ignore storage errors */ }

        if (normalizedMode === 'off') {
            removeOverlay();
        } else {
            attachSliderIfPossible();
        }
        applyNativeVolumeVisibility();
    }

    /**
     * Check if the volume slider is enabled (not in Off mode).
     */
    function isOverlayEnabled() {
        return getVolumeSliderMode() !== 'off';
    }

    function isNativeVolumeReplacementEnabled() {
        return getVolumeSliderMode() === 'replace-native';
    }

    function shouldUseNativeReplacementSlot() {
        return isNativeVolumeReplacementEnabled() && getReplaceNativePlacement() === 'native';
    }

    function applyNativeVolumeVisibility() {
        const shouldHideNative = isOverlayEnabled() && isNativeVolumeReplacementEnabled();
        document.querySelectorAll('.ytp-volume-area').forEach((area) => {
            const nextDisplay = shouldHideNative ? 'none' : '';
            if (area.style.display !== nextDisplay) {
                area.style.display = nextDisplay;
            }
        });

        if (shouldHideNative) {
            const overlay = document.getElementById(OVERLAY_ID);
            const player = getPlayerContainer();
            const controlsHost = getYouTubeControlsHost(player);
            if (overlay && player) {
                placeOverlay(overlay, player, controlsHost);
            }
        }
    }

    /**
     * Ensure we always get the active HTMLVideoElement.
     */
    function getVideoElement() {
        if (cachedVideo && cachedVideo.isConnected) {
            return cachedVideo;
        }
        const videos = document.querySelectorAll('video');
        if (!videos || videos.length === 0) {
            cachedVideo = null;
            return null;
        }
        // Prefer the largest video element (usually the main player)
        let best = videos[0];
        let bestArea = best.clientWidth * best.clientHeight;
        videos.forEach((v) => {
            const area = v.clientWidth * v.clientHeight;
            if (area > bestArea) {
                best = v;
                bestArea = area;
            }
        });
        cachedVideo = best;
        return cachedVideo;
    }

    /**
     * Find a suitable container to host the overlay.
     */
    function getPlayerContainer(video = getVideoElement()) {
        const moviePlayer = document.getElementById('movie_player');
        if (moviePlayer) {
            return moviePlayer;
        }
        return video ? video.parentElement : null;
    }

    function getYouTubeControlsHost(player) {
        if (!player) return null;
        return player.querySelector('.ytp-left-controls') || document.querySelector('.ytp-left-controls');
    }

    function getNativeVolumeArea(controlsHost) {
        return controlsHost?.querySelector?.('.ytp-volume-area') || document.querySelector('.ytp-volume-area');
    }

    function ensurePlayerPositioning(player) {
        if (!player) return;
        const style = window.getComputedStyle(player);
        if (style.position === 'static') {
            player.style.position = 'relative';
        }
    }

    function positionOverlayAboveScrubber(overlay, player) {
        const progressBar = document.querySelector('.ytp-progress-bar-padding');
        const barEl = progressBar ? (progressBar.closest('.ytp-progress-bar') || progressBar.parentElement) : null;
        const ref = barEl || progressBar;
        if (!overlay || !player || !ref) {
            if (overlay) overlay.style.bottom = '12px';
            return;
        }

        const playerRect = player.getBoundingClientRect();
        const barRect = ref.getBoundingClientRect();
        const bottomPx = playerRect.bottom - barRect.top + 8;
        overlay.style.bottom = `${Math.max(20, bottomPx)}px`;
    }

    function updateVideoOverlayPosition(overlay, player) {
        if (!overlay || !player || !isSliderOnVideo()) return;
        if (player.classList.contains('ytp-autohide')) {
            overlay.style.bottom = '12px';
            return;
        }
        positionOverlayAboveScrubber(overlay, player);
    }

    function placeOverlayInControls(overlay, controlsHost) {
        if (!overlay || !controlsHost) return;
        const nativeVolumeArea = getNativeVolumeArea(controlsHost);
        if (shouldUseNativeReplacementSlot() && nativeVolumeArea && nativeVolumeArea.parentElement === controlsHost) {
            if (overlay.parentElement !== controlsHost || overlay.nextElementSibling !== nativeVolumeArea) {
                controlsHost.insertBefore(overlay, nativeVolumeArea);
            }
        } else if (overlay.parentElement !== controlsHost || overlay.nextElementSibling) {
            controlsHost.appendChild(overlay);
        }
    }

    function placeOverlay(overlay, player, controlsHost) {
        if (!overlay || !player) return;

        if (isSliderOnVideo()) {
            ensurePlayerPositioning(player);
            if (overlay.parentElement !== player) {
                player.appendChild(overlay);
            }
            updateVideoOverlayPosition(overlay, player);
        } else if (controlsHost) {
            placeOverlayInControls(overlay, controlsHost);
            overlay.style.bottom = '';
        }

        const expanded = overlay.classList.contains('tm-expanded') || isAlwaysExpandedEnabled();
        setOverlayExpanded(overlay, expanded, true);
        updateOverlayOpacity(overlay);
    }

    function updateSliderBar(slider) {
        const value = Number(slider.value) || 0;
        const pct = Math.min(Math.max(value, 0), 100);
        const fadeStart = Math.max(0, pct - 1);
        const fadeEnd = Math.min(100, pct + 1);
        slider.style.background = `linear-gradient(to right,
            #cc4444 0%,
            #bb3333 ${fadeStart}%,
            #cc5555 ${pct}%,
            rgba(255, 255, 255, 0.15) ${fadeEnd}%,
            rgba(255, 255, 255, 0.15) 100%)`;
        slider.style.backgroundSize = 'calc(100% - var(--tm-thumb-size, 22px)) var(--tm-active-track-h, 9px)';
        slider.style.backgroundPosition = 'center';
        slider.style.backgroundRepeat = 'no-repeat';
    }

    function updateVolumeIndicator(overlay, value, muted) {
        if (!overlay) return;
        const pct = muted ? 0 : Math.min(Math.max(Number(value) || 0, 0), 100);
        overlay.title = muted ? 'Muted' : `Volume ${Math.round(pct)}%`;
        overlay.setAttribute('aria-label', overlay.title);

        const indicator = overlay.querySelector('.tm-volume-indicator');
        if (indicator) {
            indicator.classList.toggle('muted', muted);
            indicator.dataset.volumeIcon = getSpeakerIconMode(pct, muted);
        }
        const arc = overlay.querySelector('.tm-volume-arc');
        if (arc) {
            arc.style.strokeDasharray = `${pct} 100`;
        }
    }

    function setOverlayExpanded(overlay, expanded, force = false) {
        if (!overlay) return;
        if (!expanded && isAlwaysExpandedEnabled()) {
            expanded = true;
        }
        if (!force && (
            (expanded && overlay.classList.contains('tm-expanded')) ||
            (!expanded && overlay.classList.contains('tm-collapsed'))
        )) {
            return;
        }
        const onVideo = isSliderOnVideo();
        const baseStyle = onVideo ? {
            position: 'absolute',
            left: '50%',
            zIndex: '10000',
            margin: '0',
            alignSelf: 'auto',
            flex: '0 0 auto',
            transition: 'opacity 0.18s ease, width 0.22s cubic-bezier(0.16, 1, 0.3, 1), bottom 0.25s ease'
        } : {
            position: 'relative',
            left: 'auto',
            bottom: '',
            zIndex: '2',
            margin: '0 4px',
            alignSelf: 'center',
            flex: '0 0 auto',
            transition: 'width 0.22s cubic-bezier(0.16, 1, 0.3, 1)'
        };
        const pillStyle = {
            ...baseStyle,
            minWidth: '0',
            maxWidth: 'none',
            height: '40px',
            minHeight: '40px',
            borderRadius: '20px',
            flexDirection: 'row',
            alignItems: 'center',
            justifyContent: 'flex-start',
            gap: '0',
            background: 'transparent',
            transform: onVideo ? 'translateX(-50%)' : 'translateY(0)'
        };

        if (expanded) {
            overlay.classList.remove('tm-collapsed');
            overlay.classList.add('tm-expanded');
            Object.assign(overlay.style, pillStyle, {
                width: 'clamp(320px, 34vw, 460px)',
                padding: '0 12px 0 0'
            });
            updateOverlayOpacity(overlay);
            return;
        }

        overlay.classList.remove('tm-expanded');
        overlay.classList.add('tm-collapsed');
        Object.assign(overlay.style, pillStyle, {
            width: '40px',
            padding: '0'
        });
        updateOverlayOpacity(overlay);
    }

    function shouldKeepOverlayExpanded(overlay) {
        return isAlwaysExpandedEnabled() ||
            overlay.matches(':hover') ||
            overlay.contains(document.activeElement) ||
            overlay.dataset.tmDragging === 'true';
    }

    function areYouTubeControlsHidden() {
        const player = getPlayerContainer();
        return !!player?.classList?.contains('ytp-autohide') ||
            !!player?.classList?.contains('ytp-hide-controls');
    }

    function clearExpandedHold(overlay) {
        overlay.dataset.tmKeepExpanded = 'false';
        if (overlay._expandedHoldTimer) {
            clearInterval(overlay._expandedHoldTimer);
            overlay._expandedHoldTimer = 0;
        }
    }

    function markVolumeChangedWhileExpanded(overlay) {
        if (!overlay) return;
        if (isAlwaysExpandedEnabled()) {
            setOverlayExpanded(overlay, true);
            return;
        }
        overlay.dataset.tmKeepExpanded = 'true';
        setOverlayExpanded(overlay, true);
        if (overlay._expandedHoldTimer) return;

        overlay._expandedHoldTimer = window.setInterval(() => {
            if (!overlay.isConnected || areYouTubeControlsHidden()) {
                clearExpandedHold(overlay);
                setOverlayExpanded(overlay, false);
            }
        }, 250);
    }

    function collapseOverlayIfIdle(overlay, force = false) {
        window.setTimeout(() => {
            if (isAlwaysExpandedEnabled()) {
                setOverlayExpanded(overlay, true);
                return;
            }
            if (overlay.dataset.tmKeepExpanded === 'true' && !areYouTubeControlsHidden()) {
                return;
            }
            if (overlay.dataset.tmKeepExpanded === 'true') {
                clearExpandedHold(overlay);
            }
            if (force || !shouldKeepOverlayExpanded(overlay)) {
                setOverlayExpanded(overlay, false);
            }
        }, 0);
    }

    /**
     * Get the YouTube player API object if available.
     */
    function getYouTubePlayer() {
        if (cachedYtPlayer && cachedYtPlayer.isConnected && typeof cachedYtPlayer.getVolume === 'function') {
            return cachedYtPlayer;
        }
        const player = document.getElementById('movie_player');
        if (player && typeof player.getVolume === 'function') {
            cachedYtPlayer = player;
            return player;
        }
        cachedYtPlayer = null;
        return null;
    }

    /**
     * Get volume from YouTube's player API (0-100), falling back to video element.
     */
    function getVolume(video) {
        const ytPlayer = getYouTubePlayer();
        if (ytPlayer) {
            const isMuted = ytPlayer.isMuted();
            if (isMuted) {
                return 0;
            }
            return ytPlayer.getVolume();
        }
        // Fallback to video element
        const vol = (video.muted ? 0 : video.volume) || 0;
        return Math.round(vol * 100);
    }

    /**
     * Check muted state via YouTube API or video element.
     */
    function isMuted(video) {
        const ytPlayer = getYouTubePlayer();
        if (ytPlayer && typeof ytPlayer.isMuted === 'function') {
            return ytPlayer.isMuted();
        }
        return !!video.muted;
    }

    /**
     * Toggle mute via YouTube API or video element.
     */
    function toggleMute(video) {
        const ytPlayer = getYouTubePlayer();
        if (ytPlayer) {
            if (ytPlayer.isMuted()) {
                ytPlayer.unMute();
            } else {
                ytPlayer.mute();
            }
        } else {
            video.muted = !video.muted;
        }
    }

    /**
     * Set volume using YouTube's player API (0-100), falling back to video element.
     */
    function setVolume(video, value) {
        const ytPlayer = getYouTubePlayer();
        if (ytPlayer) {
            if (ytPlayer.isMuted()) {
                ytPlayer.unMute();
            }
            ytPlayer.setVolume(value);
        } else {
            // Fallback to video element
            video.volume = value / 100;
            video.muted = false;
        }
    }

    /**
     * Read saved volume from localStorage. Returns 0-100 or null.
     */
    function getSavedVolume() {
        try {
            const saved = localStorage.getItem(STORAGE_KEY);
            if (saved === null) return null;
            const value = Math.min(100, Math.max(0, parseInt(saved, 10)));
            return isNaN(value) ? null : value;
        } catch (e) { return null; }
    }

    /**
     * Restore volume from localStorage and apply to player. Call before syncing slider.
     */
    function restoreSavedVolume(video) {
        const value = getSavedVolume();
        if (value !== null) {
            setVolume(video, value);
        }
    }

    /**
     * Snap value to nearest 5% (5, 10, 15, ... 100).
     */
    function snapTo5(value) {
        return Math.round((Math.min(100, Math.max(0, value)) / 5)) * 5;
    }

    function readSnappedSliderValue(slider) {
        let value = Number(slider.value) || 0;
        if (isSnapTo5Enabled()) {
            value = snapTo5(value);
            slider.value = String(value);
        }
        return value;
    }

    /**
     * Save volume to localStorage for persistence across page reloads.
     */
    function saveVolume(value) {
        try {
            localStorage.setItem(STORAGE_KEY, String(Math.round(value)));
        } catch (e) { /* ignore storage errors */ }
    }

    function scheduleSaveVolume(value) {
        if (saveVolumeTimer) {
            clearTimeout(saveVolumeTimer);
        }
        saveVolumeTimer = window.setTimeout(() => {
            saveVolume(value);
            saveVolumeTimer = 0;
        }, STORAGE_WRITE_DEBOUNCE_MS);
    }

    /**
     * Sync mute button icon (style toggles only, no DOM mutation).
     */
    function getSpeakerIconMode(value, muted) {
        const pct = Math.min(Math.max(Number(value) || 0, 0), 100);
        if (muted || pct === 0) return 'muted';
        return pct <= 50 ? 'low' : 'high';
    }

        const panel = menu.closest('.ytp-panel');

    function isSettingsMenuToggleItem(item) {
        const role = item.getAttribute('role');
        if (role === 'menuitemcheckbox') return true;
        if (item.hasAttribute('aria-checked')) return true;
        return !!item.querySelector('.ytp-switch-module, .ytp-switch');
    }

    function isSettingsMenuSubmenuItem(item) {
        if (item.id === 'tm-volume-settings-root') return false;
        return item.getAttribute('aria-haspopup') === 'true';
    }

    function isSettingsRootMenu(menu) {
        if (!menu || menu.closest('.tm-volume-settings-submenu')) return false;
        const items = getSettingsMenuItems(menu);
        if (items.length < 3) return false;
        const toggleCount = items.filter(isSettingsMenuToggleItem).length;
        const submenuCount = items.filter(isSettingsMenuSubmenuItem).length;
        return toggleCount >= 1 && submenuCount >= 2;
    }

    function findSettingsRootMenu() {
        const popup = findOpenSettingsPopup();
        const menus = popup
            ? [popup.querySelector('.ytp-panel:not(.tm-volume-settings-submenu) > .ytp-panel-menu')].filter(Boolean)
            : Array.from(document.querySelectorAll('.ytp-settings-menu .ytp-panel-menu'));
        return menus.find(isSettingsRootMenu) || null;
    }

    function createSettingsItem(id, label, contentText, onActivate, hasSubmenu = false, iconMode = 'speaker') {
        const item = document.createElement('div');
        item.id = id;
        item.className = 'ytp-menuitem tm-volume-settings-item';
        item.setAttribute('role', 'menuitem');
        if (hasSubmenu) item.setAttribute('aria-haspopup', 'true');
        item.tabIndex = 0;

        const iconEl = document.createElement('div');
        iconEl.className = 'ytp-menuitem-icon tm-volume-settings-icon';
        if (iconMode === 'speaker') {
            iconEl.appendChild(createSettingsIconSvg());
        }

        const labelEl = document.createElement('div');
        labelEl.className = 'ytp-menuitem-label tm-volume-settings-label';
        labelEl.textContent = label;

        const contentEl = document.createElement('div');
        contentEl.className = 'ytp-menuitem-content tm-volume-settings-value';
        contentEl.textContent = contentText || '';

        const activate = (event) => {
            event.preventDefault();
            event.stopPropagation();
            window.setTimeout(onActivate, 0);
        };

        item.addEventListener('click', activate);
        item.addEventListener('keydown', (event) => {
            if (event.key === 'Enter' || event.key === ' ') {
                activate(event);
            }
        });

        item.appendChild(iconEl);
        item.appendChild(labelEl);
        if (iconMode !== 'check' || contentText) {
            item.appendChild(contentEl);
        }
        return item;
    }

    function measureSettingsMenuWidth(menu) {
        const { chain } = getSettingsMenuPopupElements(menu);
        const previous = chain.map((el) => ({
            el,
            width: el.style.width,
            maxWidth: el.style.maxWidth,
        }));

        previous.forEach(({ el }) => {
            el.style.width = 'auto';
            el.style.maxWidth = 'none';
        });

        const itemWidths = getSettingsMenuItems(menu).map((item) => Math.max(
            item.scrollWidth,
            item.getBoundingClientRect().width
        ));
        const measured = Math.ceil(Math.max(menu.scrollWidth, menu.getBoundingClientRect().width, ...itemWidths));

        previous.forEach(({ el, width, maxWidth }) => {
            el.style.width = width;
            el.style.maxWidth = maxWidth;
        });

        return Math.max(300, measured + 12);
    }

    function fitSettingsPopupToMenu(menu) {
        const { panel, chain } = getSettingsMenuPopupElements(menu);
        const contentHeight = getSettingsMenuItems(menu).reduce((height, child) => {
            const rect = child.getBoundingClientRect();
            return height + (rect.height || child.offsetHeight || 40);
        }, 0);
        const header = panel?.querySelector(':scope > .ytp-panel-header');
        const headerHeight = header ? (header.getBoundingClientRect().height || header.offsetHeight || 0) : 0;
        const nextHeight = Math.ceil(Math.max(contentHeight + headerHeight, panel?.scrollHeight || 0));
        const isVolumeSubmenu = panel?.classList?.contains('tm-volume-settings-submenu');
        const nextWidth = isVolumeSubmenu
            ? 251
            : Math.round(measureSettingsMenuWidth(menu));

        chain.forEach((el) => {
            el.style.height = `${nextHeight}px`;
            el.style.maxHeight = `${nextHeight}px`;
            el.style.overflow = 'hidden';
            el.style.width = `${nextWidth}px`;
            el.style.maxWidth = `${nextWidth}px`;
        });
        if (header) {
            menu.style.height = `${Math.max(0, nextHeight - headerHeight)}px`;
        }
    }

    function ensureSettingsControlsObserver() {
        const player = getPlayerContainer();
        if (!player) return;
        if (window.__tmYtVolumeSettingsControlsObserver &&
            window.__tmYtVolumeSettingsControlsObserverTarget === player &&
            player.isConnected) {
            return;
        }

        if (window.__tmYtVolumeSettingsControlsObserver) {
            window.__tmYtVolumeSettingsControlsObserver.disconnect();
            window.__tmYtVolumeSettingsControlsObserver = null;
            window.__tmYtVolumeSettingsControlsObserverTarget = null;
        }

        window.__tmYtVolumeSettingsControlsObserver = new MutationObserver(() => {
            resetVolumeSettingsSubmenuIfControlsHidden();
        });
        window.__tmYtVolumeSettingsControlsObserverTarget = player;
        window.__tmYtVolumeSettingsControlsObserver.observe(player, {
            attributes: true,
            attributeFilter: ['class']
        });
    }

        submenuPanel.className = 'ytp-panel tm-volume-settings-submenu';

        const header = document.createElement('div');
        header.className = 'ytp-panel-header';
        const backContainer = document.createElement('div');
        backContainer.className = 'ytp-panel-back-button-container';
        const backButton = document.createElement('button');
        backButton.className = 'ytp-button ytp-panel-back-button';
        backButton.setAttribute('aria-label', 'Back to previous menu');
        backContainer.appendChild(backButton);
        const headerTitle = document.createElement('span');
        headerTitle.className = 'ytp-panel-title';
        headerTitle.setAttribute('role', 'button');
        headerTitle.tabIndex = 0;
        headerTitle.title = 'Back to previous menu';
        headerTitle.setAttribute('aria-label', 'Back to previous menu');
        headerTitle.textContent = 'Volume Slider';
        header.appendChild(backContainer);
        header.appendChild(headerTitle);
        const back = (event) => {
            event.preventDefault();
            event.stopPropagation();
            closeVolumeSettingsSubmenu();
        };
        bindSubmenuBackTarget(backButton, back);
        bindSubmenuBackTarget(headerTitle, back);

        const menu = document.createElement('div');
        menu.className = 'ytp-panel-menu';
        menu.setAttribute('role', 'menu');
        menu.appendChild(createVolumeSubmenuItem('On', 'on'));
        menu.appendChild(createVolumeSubmenuItem('Off', 'off'));
        menu.appendChild(createVolumeSubmenuItem('Replace native', 'replace-native'));
        menu.appendChild(createVolumeSubmenuDivider());
        menu.appendChild(createVolumeSubmenuLabel('Position when replacing'));
        menu.appendChild(createReplacementPlacementSubmenuItem('Native spot', 'native'));
        menu.appendChild(createReplacementPlacementSubmenuItem('Custom spot', 'custom'));
        menu.appendChild(createVolumeSubmenuDivider());
        menu.appendChild(createVolumeSubmenuLabel('Slider behavior'));
        menu.appendChild(createSliderBehaviorSubmenuItem('Free drag', false));
        menu.appendChild(createSliderBehaviorSubmenuItem('Snap to 5%', true));
        menu.appendChild(createAlwaysExpandedSubmenuItem());
        menu.appendChild(createVideoLocationSubmenuItem());

        submenuPanel.appendChild(header);
        submenuPanel.appendChild(menu);
        content.appendChild(submenuPanel);
        fitSettingsPopupToMenu(menu);

        mainPanel.style.display = 'none';
    }
    function injectSettingsMenuItems() {
        ensureSettingsStyles();
        const menu = findSettingsRootMenu();
        if (!menu) return false;

        const existingRoot = document.getElementById('tm-volume-settings-root');
        if (existingRoot && existingRoot.parentElement === menu) {
            updateVolumeSettingsItem();
            fitSettingsPopupToMenu(menu);
            return true;
        }

        existingRoot?.remove();
        const rootItem = createVolumeSettingsItem();
        menu.insertBefore(rootItem, findVolumeSettingsInsertionPoint(menu));
        fitSettingsPopupToMenu(menu);
        return true;
    }

    function setupSettingsMenuHandler() {
        if (window.__tmYtVolumeSettingsBound) return;
        window.__tmYtVolumeSettingsBound = true;

        document.addEventListener('click', (event) => {
            if (event.target?.closest?.('.ytp-settings-button')) {
                ensureSettingsControlsObserver();
                window.setTimeout(injectSettingsMenuItems, 0);
                window.setTimeout(injectSettingsMenuItems, 100);
            }
        }, false);

        ensureSettingsControlsObserver();
    }

    function createOverlay(video, player, controlsHost) {
        // Prevent duplicates
        const existing = document.getElementById(OVERLAY_ID);
        if (existing) {
            return existing;
        }

        createStylesIfNeeded();

        const overlay = document.createElement('div');
        overlay.id = OVERLAY_ID;
        overlay.className = 'tm-collapsed';
        overlay.tabIndex = 0;

        Object.assign(overlay.style, {
            position: 'relative',
            transform: 'translateY(0)',
            width: '40px',
            minWidth: '0',
            maxWidth: 'none',
            height: '40px',
            minHeight: '40px',
            padding: '0',
            background: 'transparent',
            backdropFilter: 'none',
            WebkitBackdropFilter: 'none',
            borderRadius: '20px',
            border: 'none',
            display: 'flex',
            flexDirection: 'row',
            alignItems: 'center',
            justifyContent: 'flex-start',
            gap: '0',
            zIndex: '2',
            pointerEvents: 'auto',
            boxSizing: 'border-box',
            overflow: 'hidden',
            opacity: '1',
            flex: '0 0 auto',
            margin: '0 4px',
            alignSelf: 'center',
            transition: 'width 0.22s cubic-bezier(0.16, 1, 0.3, 1)',
            boxShadow: 'none'
        });

        overlay.addEventListener('mouseenter', () => {
            setOverlayExpanded(overlay, true);
            updateOverlayOpacity(overlay);
        });
        overlay.addEventListener('mouseleave', () => {
            overlay.dataset.tmDragging = 'false';
            updateOverlayOpacity(overlay);
            collapseOverlayIfIdle(overlay, true);
        });
        overlay.addEventListener('focusin', () => {
            setOverlayExpanded(overlay, true);
            updateOverlayOpacity(overlay);
        });
        overlay.addEventListener('focusout', () => {
            updateOverlayOpacity(overlay);
            collapseOverlayIfIdle(overlay);
        });

        const iconCell = document.createElement('div');
        iconCell.className = 'tm-volume-icon-cell';
        const indicator = document.createElement('div');
        indicator.className = 'tm-volume-indicator';
        indicator.appendChild(makeVolumeIndicatorSvg());
        iconCell.appendChild(indicator);
        iconCell.addEventListener('click', (e) => {
            e.stopPropagation();
            e.preventDefault();
            toggleMute(video);
            setSliderFromPlayer(slider, label, video);
            markVolumeChangedWhileExpanded(overlay);
        });
        const panelBg = document.createElement('div');
        panelBg.className = 'tm-volume-panel-bg';

        const topRow = document.createElement('div');
        topRow.className = 'tm-volume-controls tm-volume-top-row';

        const label = document.createElement('div');
        label.id = VALUE_LABEL_ID;
        label.textContent = '100%';

        topRow.appendChild(label);

        const sliderWrap = document.createElement('div');
        sliderWrap.className = 'tm-volume-controls tm-volume-slider-row';
        sliderWrap.style.position = 'relative';
        sliderWrap.style.width = 'auto';
        sliderWrap.style.flex = '1 1 auto';
        sliderWrap.style.height = '40px';
        sliderWrap.style.display = 'flex';
        sliderWrap.style.alignItems = 'center';

        const tickOverlay = document.createElement('div');
        tickOverlay.className = 'tm-slider-ticks';

        const slider = document.createElement('input');
        slider.id = SLIDER_ID;
        slider.type = 'range';
        slider.min = '0';
        slider.max = '100';
        slider.step = '1';
        slider.style.width = '100%';
        slider.style.display = 'block';
        slider.style.margin = '0';
        slider.style.cursor = 'pointer';
        slider.style.backgroundSize = 'calc(100% - var(--tm-thumb-size, 22px)) 100%';
        slider.setAttribute('aria-label', 'Volume');

        // Initialize from localStorage to avoid the 100% to actual jump on video load
        const initVol = getSavedVolume();
        if (initVol !== null) {
            slider.value = String(initVol);
            label.textContent = `${initVol}%`;
        } else {
            setSliderFromPlayer(slider, label, video);
        }
        updateSliderBar(slider);

        let pointerStartX = 0;
        let pointerStartY = 0;
        let pointerStartValue = 0;
        let pointerMoved = false;
        let clickSnapHandled = false;

        const applySliderValue = (value) => {
            setVolume(video, value);
            label.textContent = `${value}%`;
            updateSliderBar(slider);
            updateVolumeIndicator(overlay, value, isMuted(video));
            scheduleSaveVolume(value);
            markVolumeChangedWhileExpanded(overlay);
        };

        const snapDirectClickIfNeeded = () => {
            const currentValue = Number(slider.value) || 0;
            if (clickSnapHandled || pointerMoved || currentValue === pointerStartValue) return;
            const snappedValue = snapTo5(currentValue);
            slider.value = String(snappedValue);
            applySliderValue(snappedValue);
            clickSnapHandled = true;
        };

        const finishSliderInteraction = (event) => {
            if (event?.type === 'pointerup') {
                snapDirectClickIfNeeded();
            }
            overlay.dataset.tmDragging = 'false';
            updateOverlayOpacity(overlay);
            collapseOverlayIfIdle(overlay, !overlay.matches(':hover'));
        };
        slider.addEventListener('pointerdown', (event) => {
            pointerStartX = event.clientX;
            pointerStartY = event.clientY;
            pointerStartValue = Number(slider.value) || 0;
            pointerMoved = false;
            clickSnapHandled = false;
            overlay.dataset.tmDragging = 'true';
            setOverlayExpanded(overlay, true);
            updateOverlayOpacity(overlay);
        });
        slider.addEventListener('pointermove', (event) => {
            if (Math.abs(event.clientX - pointerStartX) > 3 || Math.abs(event.clientY - pointerStartY) > 3) {
                pointerMoved = true;
            }
        });
        slider.addEventListener('pointerup', finishSliderInteraction);
        slider.addEventListener('click', snapDirectClickIfNeeded);
        slider.addEventListener('pointercancel', finishSliderInteraction);
        window.addEventListener('pointerup', finishSliderInteraction, true);
        window.addEventListener('pointercancel', finishSliderInteraction, true);
        window.addEventListener('blur', finishSliderInteraction);

        slider.addEventListener('input', () => {
            applySliderValue(readSnappedSliderValue(slider));
        });

        slider.addEventListener('change', () => {
            applySliderValue(readSnappedSliderValue(slider));
            if (saveVolumeTimer) {
                clearTimeout(saveVolumeTimer);
                saveVolumeTimer = 0;
            }
            saveVolume(Number(slider.value) || 0);
        });

        const onVideoVolumeChange = () => {
            if (!document.getElementById(OVERLAY_ID)) return;
            setSliderFromPlayer(slider, label, video);
            if (!isMuted(video)) scheduleSaveVolume(getVolume(video));
        };
        video.addEventListener('volumechange', onVideoVolumeChange);

        sliderWrap.appendChild(slider);
        sliderWrap.appendChild(tickOverlay);
        overlay.appendChild(panelBg);
        overlay.appendChild(iconCell);
        overlay.appendChild(topRow);
        overlay.appendChild(sliderWrap);

        placeOverlay(overlay, player, controlsHost);
        setSliderFromPlayer(slider, label, video);
        setOverlayExpanded(overlay, false, true);
        updateVideoOverlayPosition(overlay, player);
        const onLayoutChange = () => placeOverlay(overlay, player, controlsHost);
        window.addEventListener('resize', onLayoutChange);
        const controlsObserver = new MutationObserver(onLayoutChange);
        controlsObserver.observe(player, { attributes: true, attributeFilter: ['class'] });
        overlay._volumeCleanup = () => {
            video.removeEventListener('volumechange', onVideoVolumeChange);
            window.removeEventListener('pointerup', finishSliderInteraction, true);
            window.removeEventListener('pointercancel', finishSliderInteraction, true);
            window.removeEventListener('blur', finishSliderInteraction);
            window.removeEventListener('resize', onLayoutChange);
            controlsObserver.disconnect();
            clearExpandedHold(overlay);
        };

        return overlay;
    }

    function removeOverlay() {
        if (saveVolumeTimer) {
            clearTimeout(saveVolumeTimer);
            saveVolumeTimer = 0;
        }
        const existing = document.getElementById(OVERLAY_ID);
        if (existing && existing.parentElement) {
            if (typeof existing._volumeCleanup === 'function') {
                existing._volumeCleanup();
            }
            existing.parentElement.removeChild(existing);
        }
        applyNativeVolumeVisibility();
    }

    function attachSliderIfPossible() {
        // Only create overlay if enabled
        if (!isOverlayEnabled()) {
            applyNativeVolumeVisibility();
            return false;
        }

        const video = getVideoElement();
        const player = getPlayerContainer(video);
        const controlsHost = getYouTubeControlsHost(player);
        if (!video || !player || (!controlsHost && !isSliderOnVideo())) {
            return false;
        }
        const existingOverlay = document.getElementById(OVERLAY_ID);
        if (existingOverlay) {
            placeOverlay(existingOverlay, player, controlsHost);
        } else {
            restoreSavedVolume(video);
            createOverlay(video, player, controlsHost);
        }
        applyNativeVolumeVisibility();
        return true;
    }

    function setupInitialAttempts() {
        let attempts = 0;
        const maxAttempts = 20; // ~10 seconds with 500ms interval
        const interval = setInterval(() => {
            attempts++;
            if (attachSliderIfPossible() || attempts >= maxAttempts) {
                clearInterval(interval);
            }
        }, 500);
    }

    function setupMutationObserver() {
        if (!window.MutationObserver) return;

        let lastAttach = 0;
        const ATTACH_COOLDOWN_MS = 1000;
        let attachQueued = false;

        const observer = new MutationObserver((mutations) => {
            if (document.location.pathname !== '/watch') return;
            const hasAddedNodes = mutations.some((m) => m.addedNodes && m.addedNodes.length > 0);

            // Slider reattach (only when new nodes appear)
            if (hasAddedNodes && !attachQueued) {
                const existing = document.getElementById(OVERLAY_ID);
                if (!existing || !existing.isConnected) {
                    attachQueued = true;
                    requestAnimationFrame(() => {
                        attachQueued = false;
                        const now = Date.now();
                        if (now - lastAttach < ATTACH_COOLDOWN_MS) return;
                        if (document.location.pathname !== '/watch') return;
                        if (attachSliderIfPossible()) {
                            lastAttach = now;
                        }
                    });
                }
            }
        });

        observer.observe(document.body, {
            childList: true,
            subtree: true
        });
    }

    function setupYtNavigationHandler() {
        if (window.__tmYtVolumeNavBound) return;
        window.__tmYtVolumeNavBound = true;

        const runReattach = () => {
            if (navReattachTimer) {
                clearTimeout(navReattachTimer);
                navReattachTimer = 0;
            }

            cachedVideo = null;
            cachedYtPlayer = null;

            navReattachTimer = window.setTimeout(() => {
                navReattachTimer = 0;
                removeOverlay();
                attachSliderIfPossible();
            }, NAV_REATTACH_DELAY_MS);
        };

        const scheduleReattach = () => {
            if (navDebounceTimer) {
                clearTimeout(navDebounceTimer);
                navDebounceTimer = 0;
            }
            navDebounceTimer = window.setTimeout(() => {
                navDebounceTimer = 0;
                runReattach();
            }, NAV_DEBOUNCE_MS);
        };

        window.addEventListener('yt-navigate-finish', scheduleReattach, true);
        window.addEventListener('yt-page-data-updated', scheduleReattach, true);
    }

    function init() {
        setupInitialAttempts();
        setupMutationObserver();
        setupYtNavigationHandler();
        setupSettingsMenuHandler();
    }

    if (document.readyState === 'complete' || document.readyState === 'interactive') {
        init();
    } else {
        window.addEventListener('DOMContentLoaded', init, { once: true });
    }
})();
