    /**
     * Check if overlay is enabled from localStorage (default: true).
     */
    function isOverlayEnabled() {
        try {
            const saved = localStorage.getItem(OVERLAY_ENABLED_KEY);
            if (saved === null) return true; // default enabled
            return saved === 'true';
        } catch (e) {
            return true;
        }
    }

    /**
     * Set overlay enabled state and save to localStorage.
     */
    function setOverlayEnabled(enabled) {
        try {
            localStorage.setItem(OVERLAY_ENABLED_KEY, enabled ? 'true' : 'false');
        } catch (e) { /* ignore storage errors */ }
        
        if (enabled) {
            showOverlay();
        } else {
            hideOverlay();
        }
        updateToggleButtonState();
    }

    /**
     * Show the overlay slider.
     */
    function showOverlay() {
        attachSliderIfPossible();
    }

    /**
     * Hide the overlay slider.
     */
    function hideOverlay() {
        removeOverlay();
    }

    function applyToggleButtonVisualState(button, enabled) {
        if (!button) return;
        button.classList.toggle('tm-toggle-active', enabled);
        button.title = enabled ? 'Disable Slider' : 'Enable Slider';
        button.setAttribute('aria-label', enabled ? 'Disable Slider' : 'Enable Slider');
        // Enforce visual state inline so site CSS cannot override it.
        button.style.background = enabled ? 'rgba(255, 255, 255, 0.16)' : 'rgba(255, 255, 255, 0.05)';
        button.style.border = enabled ? '1px solid rgba(255, 255, 255, 0.38)' : '1px solid rgba(255, 255, 255, 0.16)';
        button.style.opacity = enabled ? '1' : '0.65';

        const iconOn = button.querySelector('.tm-icon-on');
        const iconOff = button.querySelector('.tm-icon-off');
        if (iconOn) iconOn.style.display = enabled ? '' : 'none';
        if (iconOff) iconOff.style.display = enabled ? 'none' : '';
    }

    /**
     * Update toggle button appearance based on overlay state.
     */
    function updateToggleButtonState() {
        if (!cachedToggleButton) return;
        applyToggleButtonVisualState(cachedToggleButton, isOverlayEnabled());
    }

    /**
     * Ensure we always get the active HTMLVideoElement.
     */
    function getVideoElement() {
        if (cachedVideo && cachedVideo.isConnected) {
            return cachedVideo;
        }
        const videos = document.querySelectorAll('video');
        if (!videos || videos.length === 0) {
            cachedVideo = null;
            return null;
        }
        // Prefer the largest video element (usually the main player)
        let best = videos[0];
        let bestArea = best.clientWidth * best.clientHeight;
        videos.forEach((v) => {
            const area = v.clientWidth * v.clientHeight;
            if (area > bestArea) {
                best = v;
                bestArea = area;
            }
        });
        cachedVideo = best;
        return cachedVideo;
    }

    /**
     * Find a suitable container to host the overlay.
     */
    function getPlayerContainer(video = getVideoElement()) {
        const moviePlayer = document.getElementById('movie_player');
        if (moviePlayer) {
            return moviePlayer;
        }
        return video ? video.parentElement : null;
    }

    function ensurePlayerPositioning(player) {
        const style = window.getComputedStyle(player);
        if (style.position === 'static') {
            player.style.position = 'relative';
        }
    }

        return USER_SETTINGS.sliderSize === 'big' ? 'big' : 'small';

    }

    }

    function updateSliderBar(slider) {
        const value = Number(slider.value) || 0;
        const pct = Math.min(Math.max(value, 0), 100);
        const fadeStart = Math.max(0, pct - 1);
        const fadeEnd = Math.min(100, pct + 1);
        slider.style.background = `linear-gradient(to right,
            #cc4444 0%,
            #bb3333 ${fadeStart}%,
            #cc5555 ${pct}%,
            rgba(255, 255, 255, 0.15) ${fadeEnd}%,
            rgba(255, 255, 255, 0.15) 100%)`;
        slider.style.backgroundSize = '100% var(--tm-active-track-h, 9px)';
        slider.style.backgroundPosition = 'center';
        slider.style.backgroundRepeat = 'no-repeat';
    }

    /**
     * Get the YouTube player API object if available.
     */
    function getYouTubePlayer() {
        if (cachedYtPlayer && cachedYtPlayer.isConnected && typeof cachedYtPlayer.getVolume === 'function') {
            return cachedYtPlayer;
        }
        const player = document.getElementById('movie_player');
        if (player && typeof player.getVolume === 'function') {
            cachedYtPlayer = player;
            return player;
        }
        cachedYtPlayer = null;
        return null;
    }

    /**
     * Get volume from YouTube's player API (0-100), falling back to video element.
     */
    function getVolume(video) {
        const ytPlayer = getYouTubePlayer();
        if (ytPlayer) {
            const isMuted = ytPlayer.isMuted();
            if (isMuted) {
                return 0;
            }
            return ytPlayer.getVolume();
        }
        // Fallback to video element
        const vol = (video.muted ? 0 : video.volume) || 0;
        return Math.round(vol * 100);
    }

    /**
     * Check muted state via YouTube API or video element.
     */
    function isMuted(video) {
        const ytPlayer = getYouTubePlayer();
        if (ytPlayer && typeof ytPlayer.isMuted === 'function') {
            return ytPlayer.isMuted();
        }
        return !!video.muted;
    }

    /**
     * Toggle mute via YouTube API or video element.
     */
    function toggleMute(video) {
        const ytPlayer = getYouTubePlayer();
        if (ytPlayer) {
            if (ytPlayer.isMuted()) {
                ytPlayer.unMute();
            } else {
                ytPlayer.mute();
            }
        } else {
            video.muted = !video.muted;
        }
    }

    /**
     * Set volume using YouTube's player API (0-100), falling back to video element.
     */
    function setVolume(video, value) {
        const ytPlayer = getYouTubePlayer();
        if (ytPlayer) {
            if (ytPlayer.isMuted()) {
                ytPlayer.unMute();
            }
            ytPlayer.setVolume(value);
        } else {
            // Fallback to video element
            video.volume = value / 100;
            video.muted = false;
        }
    }

    /**
     * Restore volume from localStorage and apply to player. Call before syncing slider.
     */
    function restoreSavedVolume(video) {
        try {
            const saved = localStorage.getItem(STORAGE_KEY);
            if (saved !== null) {
                const value = Math.min(100, Math.max(0, parseInt(saved, 10)));
                if (!isNaN(value)) {
                    setVolume(video, value);
                }
            }
        } catch (e) { /* ignore storage errors */ }
    }

    /**
     * Snap value to nearest 5% (5, 10, 15, ... 100).
     */
    function snapTo5(value) {
        return Math.round((Math.min(100, Math.max(0, value)) / 5)) * 5;
    }

    /**
     * Save volume to localStorage for persistence across page reloads.
     */
    function saveVolume(value) {
        try {
            localStorage.setItem(STORAGE_KEY, String(Math.round(value)));
        } catch (e) { /* ignore storage errors */ }
    }

    function scheduleSaveVolume(value) {
        if (saveVolumeTimer) {
            clearTimeout(saveVolumeTimer);
        }
        saveVolumeTimer = window.setTimeout(() => {
            saveVolume(value);
            saveVolumeTimer = 0;
        }, STORAGE_WRITE_DEBOUNCE_MS);
    }

    /**
     * Sync mute button icon + slider disabled state (style toggles only, no DOM mutation).
     */
    function syncMuteButton(muteBtn, slider, video) {
        if (!muteBtn) return;
        const muted = isMuted(video);
        const iconOn = muteBtn.querySelector('.tm-icon-on');
        const iconOff = muteBtn.querySelector('.tm-icon-off');
        if (iconOn) iconOn.style.display = muted ? 'none' : '';
        if (iconOff) iconOff.style.display = muted ? '' : 'none';
        muteBtn.classList.toggle('muted', muted);
        muteBtn.title = muted ? 'Unmute' : 'Mute';
    }

    function setSliderFromPlayer(slider, label, muteBtn, video) {
        const muted = isMuted(video);
        const displayValue = getVolume(video);
        slider.value = String(displayValue);
        if (label) {
            label.textContent = muted ? 'Muted' : `${displayValue}%`;
        }
        updateSliderBar(slider);
        syncMuteButton(muteBtn, slider, video);
    }

    /**
     * Position overlay just above the video scrubber bar. Adapts to any resolution.
     */
    function positionOverlayAboveScrubber(overlay, player) {
        const progressBar = document.querySelector('.ytp-progress-bar-padding');
        const barEl = progressBar ? (progressBar.closest('.ytp-progress-bar') || progressBar.parentElement) : null;
        const ref = barEl || progressBar;
        if (ref) {
            const playerRect = player.getBoundingClientRect();
            const barRect = ref.getBoundingClientRect();
            const gap = 8;
            const bottomPx = playerRect.bottom - barRect.top + gap;
            overlay.style.bottom = `${Math.max(20, bottomPx)}px`;
        } else {
            overlay.style.bottom = '15%';
        }
    }

    /**
     * Build an SVG element (no innerHTML, safe from MutationObserver feedback).
     */
    function makeSvg(paths) {
        const ns = 'http://www.w3.org/2000/svg';
        const svg = document.createElementNS(ns, 'svg');
        svg.setAttribute('width', '20');
        svg.setAttribute('height', '20');
        svg.setAttribute('viewBox', '0 0 24 24');
        svg.setAttribute('aria-hidden', 'true');
        svg.setAttribute('focusable', 'false');
        paths.forEach((d) => {
            const p = document.createElementNS(ns, 'path');
            p.setAttribute('fill', 'currentColor');
            p.setAttribute('d', d);
            svg.appendChild(p);
        });
        return svg;
    }

    function makeToggleIcon(className, paths, isVisible) {
        const icon = document.createElement('span');
        icon.className = className;
        icon.style.display = isVisible ? '' : 'none';
        icon.appendChild(makeSvg(paths));
        return icon;
    }

        button.addEventListener('click', (e) => {

            e.stopPropagation();

            applyToggleButtonVisualState(button, newState);
            setOverlayEnabled(newState);
        });
        
        cachedToggleButton = button;
        updateToggleButtonState();
        return button;
    }

            fontSize: '18px',
            fontWeight: '500',
            color: 'rgba(255, 255, 255, 0.95)',
            textAlign: 'center',
            userSelect: 'none',
            letterSpacing: '0.5px',
            flex: '1'
        });
        label.textContent = '100%';

        const rightSpacer = document.createElement('div');
        rightSpacer.style.width = '36px';
        rightSpacer.style.flexShrink = '0';

        topRow.appendChild(muteBtn);
        topRow.appendChild(label);
        topRow.appendChild(rightSpacer);

        // --- Slider ---
        const sliderWrap = document.createElement('div');
        const sliderCfg = getSliderSizeConfig(getSliderSize());
        sliderWrap.style.position = 'relative';
        sliderWrap.style.width = '100%';
        sliderWrap.style.height = sliderCfg.hitbox;
        sliderWrap.style.display = 'flex';
        sliderWrap.style.alignItems = 'center';

        const tickOverlay = document.createElement('div');
        tickOverlay.className = 'tm-slider-ticks';

        const slider = document.createElement('input');
        slider.id = SLIDER_ID;
        slider.type = 'range';
        slider.min = '0';
        slider.max = '100';
        slider.step = '1';
        slider.style.width = '100%';
        slider.style.display = 'block';
        slider.style.margin = '0';
        slider.style.cursor = 'pointer';
        slider.style.backgroundSize = '100% 100%';
        slider.setAttribute('aria-label', 'Volume');

        setSliderFromPlayer(slider, label, muteBtn, video);

        let dragStartTime = 0;
        let isDragging = false;
        slider.addEventListener('pointerdown', () => { 
            dragStartTime = Date.now();
            isDragging = false;
        });

        slider.addEventListener('input', () => {
            const rawValue = Number(slider.value) || 0;
            const timeSinceStart = Date.now() - dragStartTime;
            
            // Snap immediately for quick clicks, otherwise smooth drag
            let value = rawValue;
            if (timeSinceStart < 300 && !isDragging) {
                value = snapTo5(rawValue);
                slider.value = String(value);
            } else {
                isDragging = true;
            }
            
            setVolume(video, value); // also unmutes if muted
            label.textContent = `${value}%`;
            updateSliderBar(slider);
            scheduleSaveVolume(value);
            syncMuteButton(muteBtn, slider, video);
        });

        slider.addEventListener('change', () => {
            const value = Number(slider.value) || 0;
            const snap = isDragging ? value : snapTo5(value);
            if (snap !== value) {
                slider.value = String(snap);
                setVolume(video, snap);
                label.textContent = `${snap}%`;
                updateSliderBar(slider);
            }
            if (saveVolumeTimer) {
                clearTimeout(saveVolumeTimer);
                saveVolumeTimer = 0;
            }
            saveVolume(Number(slider.value) || 0);
            isDragging = false;
        });

        const onVideoVolumeChange = () => {
            if (!document.getElementById(OVERLAY_ID)) return;
            setSliderFromPlayer(slider, label, muteBtn, video);
            if (!isMuted(video)) scheduleSaveVolume(getVolume(video));
        };
        video.addEventListener('volumechange', onVideoVolumeChange);

        sliderWrap.appendChild(slider);
        sliderWrap.appendChild(tickOverlay);
        overlay.appendChild(topRow);
        overlay.appendChild(sliderWrap);
        player.appendChild(overlay);

        positionOverlayAboveScrubber(overlay, player);
        setTimeout(() => {
            if (!document.querySelector('.ytp-progress-bar-padding')) return;
            positionOverlayAboveScrubber(overlay, player);
        }, 500);

        const onResize = () => positionOverlayAboveScrubber(overlay, player);
        window.addEventListener('resize', onResize);
        overlay._resizeCleanup = () => window.removeEventListener('resize', onResize);
        overlay._volumeCleanup = () => {
            video.removeEventListener('volumechange', onVideoVolumeChange);
        };

        return overlay;
    }

    function removeOverlay() {
        if (saveVolumeTimer) {
            clearTimeout(saveVolumeTimer);
            saveVolumeTimer = 0;
        }
        const existing = document.getElementById(OVERLAY_ID);
        if (existing && existing.parentElement) {
            if (typeof existing._resizeCleanup === 'function') {
                existing._resizeCleanup();
            }
            if (typeof existing._volumeCleanup === 'function') {
                existing._volumeCleanup();
            }
            existing.parentElement.removeChild(existing);
        }
    }

    /**
     * Inject toggle button into YouTube's video controls.
     */
    function injectToggleButton() {
        if (!USER_SETTINGS.showToggleButton) return false;
        if (document.location.pathname !== '/watch') return false;
        
        // Check if button already exists
        const existing = document.getElementById(TOGGLE_BTN_ID);
        if (existing && existing.isConnected) return true;
        
        // Find the actions container (where Like button is)
        const actionsContainer = document.querySelector('#actions, ytd-menu-renderer.ytd-watch-metadata');
        if (!actionsContainer) return false;
        
        // Find the flex container inside
        const flexContainer = actionsContainer.querySelector('#top-level-buttons-computed, #top-level-buttons');
        if (!flexContainer) return false;
        
        // Create and insert toggle button as first child
        const toggleBtn = createToggleButton();
        flexContainer.insertBefore(toggleBtn, flexContainer.firstChild);
        
        return true;
    }

    /**
     * Setup observer to re-inject toggle button when UI changes.
     */
    function setupToggleButtonObserver() {
        if (!USER_SETTINGS.showToggleButton) return;
        
        let lastInject = 0;
        const INJECT_COOLDOWN_MS = 500;
        let injectQueued = false;
        
        const observer = new MutationObserver(() => {
            if (document.location.pathname !== '/watch') {
                // Clean up button if not on watch page
                const existing = document.getElementById(TOGGLE_BTN_ID);
                if (existing) {
                    cachedToggleButton = null;
                    existing.remove();
                }
                return;
            }
            
            const existing = document.getElementById(TOGGLE_BTN_ID);
            if (existing && existing.isConnected) return;
            
            if (injectQueued) return;
            injectQueued = true;
            
            requestAnimationFrame(() => {
                injectQueued = false;
                const now = Date.now();
                if (now - lastInject < INJECT_COOLDOWN_MS) return;
                
                if (injectToggleButton()) {
                    lastInject = now;
                }
            });
        });
        
        observer.observe(document.body, {
            childList: true,
            subtree: true
        });
    }

    function attachSliderIfPossible() {
        // Always try to inject toggle button if enabled
        if (USER_SETTINGS.showToggleButton) {
            injectToggleButton();
        }
        
        // Only create overlay if enabled
        if (!isOverlayEnabled()) {
            return false;
        }
        
        const video = getVideoElement();
        const player = getPlayerContainer(video);
        if (!video || !player) {
            return false;
        }
        // Only restore volume on first attach; skip if overlay already exists
        if (!document.getElementById(OVERLAY_ID)) {
            restoreSavedVolume(video);
            createOverlay(player, video);
        }
        return true;
    }

    function setupInitialAttempts() {
        let attempts = 0;
        const maxAttempts = 40; // ~20 seconds with 500ms interval
        const interval = setInterval(() => {
            attempts++;
            if (attachSliderIfPossible() || attempts >= maxAttempts) {
                clearInterval(interval);
            }
        }, 500);
    }

    function setupMutationObserver() {
        if (!window.MutationObserver) return;

        let lastAttach = 0;
        const ATTACH_COOLDOWN_MS = 1000;
        let attachQueued = false;

        const observer = new MutationObserver((mutations) => {
            const existing = document.getElementById(OVERLAY_ID);
            if (existing && existing.isConnected) {
                const video = getVideoElement();
                const player = getPlayerContainer(video);
                if (video && player && existing.parentElement === player) {
                    return;
                }
            }
            if (!mutations.some((mutation) => mutation.addedNodes && mutation.addedNodes.length > 0)) {
                return;
            }
            if (attachQueued) return;
            attachQueued = true;
            requestAnimationFrame(() => {
                attachQueued = false;
            const now = Date.now();
            if (now - lastAttach < ATTACH_COOLDOWN_MS) return;
            if (document.location.pathname !== '/watch') return;
            if (attachSliderIfPossible()) {
                lastAttach = now;
            }
            });
        });

        observer.observe(document.body, {
            childList: true,
            subtree: true
        });
    }

    function setupYtNavigationHandler() {
        if (window.__tmYtVolumeNavBound) return;
        window.__tmYtVolumeNavBound = true;
        const handler = () => {
            // After navigation finishes, reattach the slider and toggle button
            setTimeout(() => {
                removeOverlay();
                cachedToggleButton = null;
                attachSliderIfPossible();
            }, 700);
        };

        window.addEventListener('yt-navigate-finish', handler, true);
        window.addEventListener('yt-page-data-updated', handler, true);
    }

    function init() {
        // Only care about watch pages
        if (document.location.pathname !== '/watch') {
            // However, YouTube can change path after load; still prepare observers.
        }
        setupInitialAttempts();
        setupMutationObserver();
        setupYtNavigationHandler();
        setupToggleButtonObserver();
    }

    if (document.readyState === 'complete' || document.readyState === 'interactive') {
        init();
    } else {
        window.addEventListener('DOMContentLoaded', init, { once: true });
    }
})();
