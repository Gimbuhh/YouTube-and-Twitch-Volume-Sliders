    /**
     * Ensure we always get the active HTMLVideoElement.
     */
    function getVideoElement() {
        if (cachedVideo && cachedVideo.isConnected) {
            return cachedVideo;
        }
        const videos = document.querySelectorAll('video');
        if (!videos || videos.length === 0) {
            cachedVideo = null;
            return null;
        }
        // Prefer the largest video element (usually the main player)
        let best = videos[0];
        let bestArea = best.clientWidth * best.clientHeight;
        videos.forEach((v) => {
            const area = v.clientWidth * v.clientHeight;
            if (area > bestArea) {
                best = v;
                bestArea = area;
            }
        });
        cachedVideo = best;
        return cachedVideo;
    }

            video.parentElement

    }

    /**
     * Get Twitch's internal player API (mediaPlayerInstance) via React fiber.
     * Returns object with getVolume(), setVolume(), isMuted(), setMuted() or null.
     */
    function getTwitchPlayerApi() {
        try {
            const playerSel = 'div[data-a-target="player-overlay-click-handler"], .video-player';
            const el = document.querySelector(playerSel);
            if (!el) return null;
            if (cachedApi && cachedApiFromElement === el) {
                return cachedApi;
            }

            let instance;
            for (const key in el) {
                if (key.startsWith('__reactInternalInstance$') || key.startsWith('__reactFiber$')) {
                    instance = el[key];
                    break;
                }
            }
            if (!instance) return null;

            let parent = instance.return;
            for (let i = 0; i < 50; i++) {
                if (!parent || !parent.memoizedProps) break;
                const player = parent.memoizedProps.mediaPlayerInstance;
                if (player) {
                    const api = player.core ?? player.playerInstance?.core ?? player;
                    if (typeof api?.getVolume === 'function' && typeof api?.setVolume === 'function') {
                        cachedApi = api;
                        cachedApiFromElement = el;
                        return api;
                    }
                }
                parent = parent.return;
            }
        } catch (e) { /* React structure may change */ }
        cachedApi = null;
        cachedApiFromElement = null;
        return null;
    }

    /**
     * Get effective volume (0 when muted, else 0-100). Use for display when unmuted.
     */
    function getVolume(video) {
        try {
            const api = getTwitchPlayerApi();
            if (api) {
                if (api.isMuted && api.isMuted()) return 0;
                const vol = api.getVolume();
                return Math.round((typeof vol === 'number' ? vol : 0) * 100);
            }
        } catch (e) { /* fall through */ }
        const vol = (video.muted ? 0 : video.volume) || 0;
        return Math.round(vol * 100);
    }

    /**
     * Set volume via Twitch API (preferred) or video element from percentage (0-100).
     */
    function setVolume(video, value) {
        try {
            const api = getTwitchPlayerApi();
            if (api) {
                if (api.isMuted && api.isMuted()) api.setMuted(false);
                api.setVolume(Math.min(1, Math.max(0, value / 100)));
                return;
            }
        } catch (e) { /* fall through */ }
        video.volume = value / 100;
        video.muted = false;
    }

    /**
     * Check if muted via Twitch API or video element.
     */
    function isMuted(video) {
        try {
            const api = getTwitchPlayerApi();
            if (api && api.isMuted) return api.isMuted();
        } catch (e) { /* fall through */ }
        return !!video.muted;
    }

    /**
     * Set mute state via Twitch API or video element.
     */
    function setMuted(video, muted) {
        try {
            const api = getTwitchPlayerApi();
            if (api && api.setMuted) {
                api.setMuted(!!muted);
                return;
            }
        } catch (e) { /* fall through */ }
        video.muted = !!muted;
    }

    /**
     * Restore volume and mute from localStorage. Call before syncing slider.
     */
    function restoreSavedVolume(video) {
        try {
            const saved = localStorage.getItem(STORAGE_KEY);
            if (saved !== null) {
                const value = Math.min(100, Math.max(0, parseInt(saved, 10)));
                if (!isNaN(value)) {
                    setVolume(video, value);
                }
            }
            const savedMute = localStorage.getItem(MUTE_STORAGE_KEY);
            if (savedMute === 'true') {
                try { setMuted(video, true); } catch (e) { /* ignore */ }
            }
        } catch (e) { /* ignore storage errors */ }
    }

    /**
     * Snap value to nearest 5% (5, 10, 15, ... 100).
     */
    function snapTo5(value) {
        return Math.round((Math.min(100, Math.max(0, value)) / 5)) * 5;
    }

    /**
     * Save volume to localStorage for persistence across page reloads.
     */
    function saveVolume(value) {
        try {
            localStorage.setItem(STORAGE_KEY, String(Math.round(value)));
        } catch (e) { /* ignore storage errors */ }
    }

    function scheduleSaveVolume(value) {
        if (saveVolumeTimer) {
            clearTimeout(saveVolumeTimer);
        }
        saveVolumeTimer = window.setTimeout(() => {
            saveVolume(value);
            saveVolumeTimer = 0;
        }, STORAGE_WRITE_DEBOUNCE_MS);
    }

    /**
     * Save mute state to localStorage.
     */
    function saveMute(muted) {
        try {
            localStorage.setItem(MUTE_STORAGE_KEY, muted ? 'true' : 'false');
        } catch (e) { /* ignore storage errors */ }
    }

    function setSliderFromPlayer(slider, label, muteBtn, video) {
        try {
            const muted = isMuted(video);
            const displayValue = getVolume(video);
            slider.value = String(displayValue);
            if (label) {
                label.textContent = muted ? 'Muted' : `${displayValue}%`;
            }
            syncMuteButton(muteBtn, muted);
            updateSliderBar(slider);
        } catch (e) { /* prevent crash */ }
    }

    function syncMuteButton(muteBtn, muted) {
        if (!muteBtn) return;
        const iconOn = muteBtn.querySelector('.tm-icon-on');
        const iconOff = muteBtn.querySelector('.tm-icon-off');
        if (iconOn) iconOn.style.display = muted ? 'none' : '';
        if (iconOff) iconOff.style.display = muted ? '' : 'none';
        muteBtn.classList.toggle('muted', muted);
        muteBtn.title = muted ? 'Unmute' : 'Mute';
    }

        svg.setAttribute('width', '20');
        svg.setAttribute('height', '20');
        svg.setAttribute('viewBox', '0 0 24 24');
        svg.setAttribute('aria-hidden', 'true');

            const p = document.createElementNS(ns, 'path');
            p.setAttribute('fill', 'currentColor');
            p.setAttribute('d', d);
            svg.appendChild(p);
        });
        return svg;
    }

            const newState = !isOverlayEnabled();
            applyToggleButtonVisualState(button, newState);
            setOverlayEnabled(newState);
        });
        
        cachedToggleButton = button;
        updateToggleButtonState();
        return button;
    }

        ]));

        muteBtn.appendChild(iconOn);
        muteBtn.appendChild(iconOff);

        const label = document.createElement('div');
        label.id = VALUE_LABEL_ID;
        Object.assign(label.style, {
            fontSize: '18px',
            fontWeight: '500',
            color: 'rgba(255, 255, 255, 0.95)',
            textAlign: 'center',
            userSelect: 'none',
            letterSpacing: '0.5px',
            flex: '1'
        });
        label.textContent = '100%';

        const rightSpacer = document.createElement('div');
        rightSpacer.style.width = '36px';
        rightSpacer.style.flexShrink = '0';

        topRow.appendChild(muteBtn);
        topRow.appendChild(label);
        topRow.appendChild(rightSpacer);

        const sliderWrap = document.createElement('div');
        const sliderCfg = getSliderSizeConfig(getSliderSize());
        sliderWrap.style.position = 'relative';
        sliderWrap.style.width = '100%';
        sliderWrap.style.height = sliderCfg.hitbox;
        sliderWrap.style.display = 'flex';
        sliderWrap.style.alignItems = 'center';

        const tickOverlay = document.createElement('div');
        tickOverlay.className = 'tm-slider-ticks';

        const slider = document.createElement('input');
        slider.id = SLIDER_ID;
        slider.type = 'range';
        slider.min = '0';
        slider.max = '100';
        slider.step = '1';
        slider.style.width = '100%';
        slider.style.display = 'block';
        slider.style.margin = '0';
        slider.style.cursor = 'pointer';
        slider.style.backgroundSize = '100% 100%';
        slider.setAttribute('aria-label', 'Volume');

        // Initialize from localStorage to avoid the 100% → actual jump on stream load
        try {
            const saved = localStorage.getItem(STORAGE_KEY);
            if (saved !== null) {
                const val = Math.min(100, Math.max(0, parseInt(saved, 10)));
                if (!isNaN(val)) {
                    slider.value = String(val);
                    label.textContent = `${val}%`;
                }
            }
        } catch (e) { /* fall through to player sync */ }
        updateSliderBar(slider);

        let dragStartTime = 0;
        let isDragging = false;
        slider.addEventListener('pointerdown', () => { 
            dragStartTime = Date.now();
            isDragging = false;
        });

        muteBtn.addEventListener('click', () => {
            try {
                const muted = isMuted(video);
                setMuted(video, !muted);
                saveMute(!muted);
                setSliderFromPlayer(slider, label, muteBtn, video);
            } catch (e) { /* prevent freeze */ }
        });

        slider.addEventListener('input', () => {
            const rawValue = Number(slider.value) || 0;
            const timeSinceStart = Date.now() - dragStartTime;
            
            // Snap immediately for quick clicks, otherwise smooth drag
            let value = rawValue;
            if (timeSinceStart < 300 && !isDragging) {
                value = snapTo5(rawValue);
                slider.value = String(value);
            } else {
                isDragging = true;
            }
            
            setVolume(video, value);
            saveMute(false);
            label.textContent = `${value}%`;
            updateSliderBar(slider);
            scheduleSaveVolume(value);
        });

        slider.addEventListener('change', () => {
            const value = Number(slider.value) || 0;
            const snap = isDragging ? value : snapTo5(value);
            if (snap !== value) {
                slider.value = String(snap);
                setVolume(video, snap);
                label.textContent = `${snap}%`;
                updateSliderBar(slider);
            }
            if (saveVolumeTimer) {
                clearTimeout(saveVolumeTimer);
                saveVolumeTimer = 0;
            }
            saveVolume(Number(slider.value) || 0);
            isDragging = false;
        });

        // Sync slider UI and persist volume on any external/native change
        // During startup lock, re-apply our saved volume if Twitch's player init overrides it
        const onVideoVolumeChange = () => {
            if (Date.now() <= startupLockUntil) {
                // Actively re-apply saved volume when Twitch overrides it during player initialization
                try {
                    const saved = localStorage.getItem(STORAGE_KEY);
                    if (saved !== null) {
                        const savedValue = Math.min(100, Math.max(0, parseInt(saved, 10)));
                        if (!isNaN(savedValue) && Math.abs(getVolume(video) - savedValue) > 1) {
                            setVolume(video, savedValue);
                        }
                    }
                } catch (e) { /* ignore */ }
                return;
            }
            try {
                setSliderFromPlayer(slider, label, muteBtn, video);
                if (!isMuted(video)) {
                    scheduleSaveVolume(getVolume(video));
                }
            } catch (e) { /* prevent crash */ }
        };
        video.addEventListener('volumechange', onVideoVolumeChange);

        sliderWrap.appendChild(slider);
        sliderWrap.appendChild(tickOverlay);
        overlay.appendChild(topRow);
        overlay.appendChild(sliderWrap);
        player.appendChild(overlay);

        positionOverlayAtBottom(overlay, player);

        const onResize = () => positionOverlayAtBottom(overlay, player);
        window.addEventListener('resize', onResize);
        overlay._resizeCleanup = () => window.removeEventListener('resize', onResize);
        overlay._volumeCleanup = () => video.removeEventListener('volumechange', onVideoVolumeChange);

        return overlay;
    }

    function removeOverlay() {
        if (saveVolumeTimer) {
            clearTimeout(saveVolumeTimer);
            saveVolumeTimer = 0;
        }
        const existing = document.getElementById(OVERLAY_ID);
        if (existing && existing.parentElement) {
            if (typeof existing._resizeCleanup === 'function') {
                existing._resizeCleanup();
            }
            if (typeof existing._volumeCleanup === 'function') {
                existing._volumeCleanup();
            }
            existing.parentElement.removeChild(existing);
        }
    }

    /**
     * Inject toggle button into Twitch's video controls (to the left of Follow button).
     */
    function injectToggleButton() {
        if (!USER_SETTINGS.showToggleButton) return false;

        const existing = document.getElementById(TOGGLE_BTN_ID);
        if (existing && existing.isConnected) return true;

        const followButton = document.querySelector('[data-a-target="follow-button"], [data-a-target="unfollow-button"]');
        if (!followButton) return false;

        const followWrapper = followButton.closest('div[class*="iglnKI"]');
        if (!followWrapper || !followWrapper.parentElement) return false;

        const wrapperClone = document.createElement('div');
        wrapperClone.className = followWrapper.className;
        wrapperClone.style.marginRight = '1rem';
        wrapperClone.style.display = 'flex';
        wrapperClone.style.alignItems = 'center';

        const innerClass = followWrapper.firstElementChild && followWrapper.firstElementChild.className
            ? followWrapper.firstElementChild.className
            : 'Layout-sc-1xcs6mc-0 dCFkna';
        const innerWrapper = document.createElement('div');
        innerWrapper.className = innerClass;
        innerWrapper.style.display = 'flex';
        innerWrapper.style.alignItems = 'center';

        const toggleBtn = createToggleButton();
        // Match Follow/Unfollow button height for consistent vertical alignment.
        const followRect = followButton.getBoundingClientRect();
        if (followRect && followRect.height > 0) {
            const h = Math.round(followRect.height);
            toggleBtn.style.height = `${h}px`;
            toggleBtn.style.minWidth = `${h}px`;
            toggleBtn.style.width = `${h}px`;
            toggleBtn.style.borderRadius = `${Math.round(h / 2)}px`;
        }
        innerWrapper.appendChild(toggleBtn);
        wrapperClone.appendChild(innerWrapper);

        // The immediate follow wrapper parent can be block-layout on some Twitch layouts.
        // Force horizontal flow so our toggle sits to the LEFT instead of above.
        const followGroup = followWrapper.parentElement;
        followGroup.style.display = 'flex';
        followGroup.style.alignItems = 'center';
        followGroup.style.flexWrap = 'nowrap';

        followGroup.insertBefore(wrapperClone, followWrapper);
        return true;
    }

    /**
     * Setup observer to re-inject toggle button when UI changes.
     * Only attempts injection when Follow button is present.
     */
    function setupToggleButtonObserver() {
        if (!USER_SETTINGS.showToggleButton) return;
        
        let lastInject = 0;
        const INJECT_COOLDOWN_MS = 500;
        let injectQueued = false;
        
        const observer = new MutationObserver(() => {
            const existing = document.getElementById(TOGGLE_BTN_ID);
            if (existing && existing.isConnected) return;
            
            // Only attempt injection if Follow/Unfollow button exists
            const followButton = document.querySelector('[data-a-target="follow-button"], [data-a-target="unfollow-button"]');
            if (!followButton) return;
            
            if (injectQueued) return;
            injectQueued = true;
            
            requestAnimationFrame(() => {
                injectQueued = false;
                const now = Date.now();
                if (now - lastInject < INJECT_COOLDOWN_MS) return;
                
                if (injectToggleButton()) {
                    lastInject = now;
                }
            });
        });
        
        observer.observe(document.body, {
            childList: true,
            subtree: true
        });
    }

    function attachSliderIfPossible() {
        // Always try to inject toggle button if enabled
        if (USER_SETTINGS.showToggleButton) {
            injectToggleButton();
        }
        
        // Only create overlay if enabled
        if (!isOverlayEnabled()) {
            return false;
        }
        
        const video = getVideoElement();
        const player = getPlayerContainer(video);
        if (!video || !player) {
            return false;
        }
        // Only restore volume and schedule fade-in on first attach
        if (!document.getElementById(OVERLAY_ID)) {
            // Extend to at least 2s from now; cooperates with the longer lock
            // already set by scheduleReattach on channel navigation without shortening it
            startupLockUntil = Math.max(startupLockUntil, Date.now() + 2000);
            restoreSavedVolume(video);
            createOverlay(player, video);
            setTimeout(() => {
                restoreSavedVolume(video);
                const el = document.getElementById(OVERLAY_ID);
                if (el) el.style.opacity = '0.45';
            }, 1500);
        }
        return true;
    }

    function setupInitialAttempts() {
        let attempts = 0;
        const maxAttempts = 40; // ~20 seconds with 500ms interval
        const interval = setInterval(() => {
            attempts++;
            if (attachSliderIfPossible() || attempts >= maxAttempts) {
                clearInterval(interval);
            }
        }, 500);
    }

    function setupMutationObserver() {
        if (!window.MutationObserver) return;

        let lastAttach = 0;
        const ATTACH_COOLDOWN_MS = 1000;
        let attachQueued = false;

        const observer = new MutationObserver((mutations) => {
            if (!mutations.some((mutation) => mutation.addedNodes && mutation.addedNodes.length > 0)) {
                return;
            }
            if (attachQueued) return;
            attachQueued = true;
            requestAnimationFrame(() => {
                attachQueued = false;
            const now = Date.now();
            if (now - lastAttach < ATTACH_COOLDOWN_MS) return;
            if (attachSliderIfPossible()) {
                lastAttach = now;
            }
            });
        });

        observer.observe(document.body, {
            childList: true,
            subtree: true
        });
    }

    function setupNavigationHandler() {
        if (window.__tmTwitchVolumeNavPatched) return;
        window.__tmTwitchVolumeNavPatched = true;
        const scheduleReattach = () => {
            // Invalidate stale player API and video caches so we re-discover fresh instances
            // after Twitch rebuilds its React fiber tree on channel navigation
            cachedApi = null;
            cachedApiFromElement = null;
            cachedVideo = null;

            // Set the startup lock IMMEDIATELY (not just when the new overlay is created at 700ms).
            // On fast machines Twitch re-initializes the player in <100ms and fires volumechange
            // before our reattach timeout fires. Without this, the OLD overlay's volumechange
            // listener would run while startupLockUntil is still 0, saving Twitch's reset
            // volume (typically 100%) over our persisted value — corrupting localStorage before
            // we even get a chance to restore.
            // Use 3500ms (vs 2000ms on initial load) — channel nav takes longer for Twitch
            // to fully re-initialize its player than a full page load does.
            startupLockUntil = Date.now() + 3500;

            setTimeout(() => {
                removeOverlay();
                cachedToggleButton = null;
                attachSliderIfPossible();
            }, 700);

            // Late recovery restore: catches cases where Twitch's player finishes its own
            // volume initialization after our startup lock has already expired
            setTimeout(() => {
                const vid = getVideoElement();
                if (!vid) return;
                try {
                    const saved = localStorage.getItem(STORAGE_KEY);
                    if (saved === null) return;
                    const savedValue = Math.min(100, Math.max(0, parseInt(saved, 10)));
                    if (isNaN(savedValue)) return;
                    if (Math.abs(getVolume(vid) - savedValue) > 1) {
                        setVolume(vid, savedValue);
                        const sliderEl = document.getElementById(SLIDER_ID);
                        const labelEl = document.getElementById(VALUE_LABEL_ID);
                        if (sliderEl) { sliderEl.value = String(savedValue); updateSliderBar(sliderEl); }
                        if (labelEl) labelEl.textContent = `${savedValue}%`;
                    }
                } catch (e) { /* ignore */ }
            }, 4200);
        };

        const originalPushState = history.pushState;
        history.pushState = function () {
            const result = originalPushState.apply(this, arguments);
            scheduleReattach();
            return result;
        };

        const originalReplaceState = history.replaceState;
        history.replaceState = function () {
            const result = originalReplaceState.apply(this, arguments);
            scheduleReattach();
            return result;
        };

        window.addEventListener('popstate', scheduleReattach, true);
    }

    function init() {
        setupInitialAttempts();
        setupMutationObserver();
        setupNavigationHandler();
        setupToggleButtonObserver();
    }

    if (document.readyState === 'complete' || document.readyState === 'interactive') {
        init();
    } else {
        window.addEventListener('DOMContentLoaded', init, { once: true });
    }
})();
